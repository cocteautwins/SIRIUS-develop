33 /// Unit step function is defined to be 1 in the interstitial and 0 inside muffin-tins.
34 /** Unit step function is constructed from it's plane-wave expansion coefficients which are computed
35  * analytically:
36  * \f[
37  * \Theta({\bf r}) = \sum_{\bf G} \Theta({\bf G}) e^{i{\bf Gr}},
38  * \f]
39  * where
40  * \f[
41  * \Theta({\bf G}) = \frac{1}{\Omega} \int \Theta({\bf r}) e^{-i{\bf Gr}} d{\bf r} =
42  * \frac{1}{\Omega} \int_{\Omega} e^{-i{\bf Gr}} d{\bf r} - \frac{1}{\Omega} \int_{MT} e^{-i{\bf Gr}} d{\bf r} =
43  * \delta_{\bf G, 0} - \sum_{\alpha} \frac{1}{\Omega} \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r}
44  * \f]
45  * Integral of a plane-wave over the muffin-tin volume is taken using the spherical expansion of the plane-wave
46  * around central point \f$ \tau_{\alpha} \f$:
47  * \f[
48  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} =
49  * e^{-i{\bf G\tau_{\alpha}}} \int_{MT_{\alpha}} 4\pi \sum_{\ell m} (-i)^{\ell} j_{\ell}(Gr)
50  * Y_{\ell m}(\hat {\bf G}) Y_{\ell m}^{*}(\hat {\bf r}) r^2 \sin \theta dr d\phi d\theta
51  * \f]
52  * In the above integral only \f$ \ell=m=0 \f$ term survives. So we have:
53  * \f[
54  * \int_{MT_{\alpha}} e^{-i{\bf Gr}} d{\bf r} = 4\pi e^{-i{\bf G\tau_{\alpha}}} \Theta(\alpha, G)
55  * \f]
56  * where
57  * \f[
58  * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
59  * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
60  * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
61  * \f]
62  * are the so-called step function form factors. With this we have a final expression for the plane-wave coefficients
63  * of the unit step function:
64  * \f[
65  * \Theta({\bf G}) = \delta_{\bf G, 0} - \sum_{\alpha} \frac{4\pi}{\Omega} e^{-i{\bf G\tau_{\alpha}}}
66  * \Theta(\alpha, G)
67  * \f]
68  */

129  /// Get \f$ \Theta(\alpha, G) \f$ form factors of the step function.
130  /**
131  * \f[
132  * \Theta(\alpha, G) = \int_{0}^{R_{\alpha}} \frac{\sin(Gr)}{Gr} r^2 dr =
133  * \left\{ \begin{array}{ll} \displaystyle R_{\alpha}^3 / 3 & G=0 \\
134  * \Big( \sin(GR_{\alpha}) - GR_{\alpha}\cos(GR_{\alpha}) \Big) / G^3 & G \ne 0 \end{array} \right.
135  * \f]
136  */
137  mdarray<double, 2> get_step_function_form_factors(int num_gsh,
138  Unit_cell const& unit_cell__,
139  Gvec const& gvec__,
140  Communicator const& comm__) const
141  {
142  mdarray<double, 2> ffac(unit_cell__.num_atom_types(), num_gsh);
143 
144  splindex<block> spl_num_gvec_shells(num_gsh, comm__.size(), comm__.rank());
145 
146  #pragma omp parallel for
147  for (int igsloc = 0; igsloc < spl_num_gvec_shells.local_size(); igsloc++)
148  {
149  int igs = spl_num_gvec_shells[igsloc];
150  double G = gvec__.shell_len(igs);
151  double g3inv = (igs) ? 1.0 / std::pow(G, 3) : 0.0;
152 
153  for (int iat = 0; iat < unit_cell__.num_atom_types(); iat++)
154  {
155  double R = unit_cell__.atom_type(iat).mt_radius();
156  double GR = G * R;
157 
158  ffac(iat, igs) = (igs) ? (std::sin(GR) - GR * std::cos(GR)) * g3inv : std::pow(R, 3) / 3.0;
159  }
160  }
161 
162  int ld = unit_cell__.num_atom_types();
163  comm__.allgather(ffac.at<CPU>(), ld * spl_num_gvec_shells.global_offset(), ld * spl_num_gvec_shells.local_size());
164  return ffac;
165  }

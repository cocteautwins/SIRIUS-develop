63  inline void generate_pseudo_rho_radial_integrals()
64  {
65  PROFILE("sirius::Radial_integrals::generate_pseudo_rho_radial_integrals");
66 
67  pseudo_rho_radial_integrals_ = mdarray<Spline<double>, 1>(unit_cell_.num_atom_types());
68 
69  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
70  auto& atom_type = unit_cell_.atom_type(iat);
71  pseudo_rho_radial_integrals_(iat) = Spline<double>(grid_gmax_);
72 
73  Spline<double> rho(atom_type.radial_grid());
74  for (int ir = 0; ir < atom_type.num_mt_points(); ir++) {
75  rho[ir] = atom_type.pp_desc().total_charge_density[ir];
76  }
77  rho.interpolate();
78 
79  #pragma omp parallel for
80  for (int iq = 0; iq < grid_gmax_.num_points(); iq++) {
81  Spherical_Bessel_functions jl(0, atom_type.radial_grid(), grid_gmax_[iq]);
82 
83  pseudo_rho_radial_integrals_(iat)[iq] = sirius::inner(jl[0], rho, 0, atom_type.num_mt_points()) / fourpi;
84  }
85  pseudo_rho_radial_integrals_(iat).interpolate();
86  }
87  }

89  /// Generate radial integrals for local part of pseudopotential.
90  /** See Potential::generate_local_potential() for more details. */
91  inline void generate_vloc_radial_integrals()
92  {
93  PROFILE("sirius::Radial_integrals::generate_vloc_radial_integrals");
94 
95  vloc_radial_integrals_ = mdarray<Spline<double>, 1>(unit_cell_.num_atom_types());
96 
97  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
98  auto& atom_type = unit_cell_.atom_type(iat);
99  vloc_radial_integrals_(iat) = Spline<double>(grid_gmax_vloc_);
100 
101  #pragma omp parallel for
102  for (int iq = 0; iq < grid_gmax_vloc_.num_points(); iq++) {
103  Spline<double> s(atom_type.radial_grid());
104  if (iq == 0) {
105  for (int ir = 0; ir < atom_type.num_mt_points(); ir++) {
106  double x = atom_type.radial_grid(ir);
107  s[ir] = (x * atom_type.pp_desc().vloc[ir] + atom_type.zn()) * x;
108  }
109  vloc_radial_integrals_(iat)[iq] = s.interpolate().integrate(0);
110  } else {
111  double g = grid_gmax_vloc_[iq];
112  double g2 = std::pow(g, 2);
113  for (int ir = 0; ir < atom_type.num_mt_points(); ir++) {
114  double x = atom_type.radial_grid(ir);
115  s[ir] = (x * atom_type.pp_desc().vloc[ir] + atom_type.zn() * gsl_sf_erf(x)) * std::sin(g * x);
116  }
117  vloc_radial_integrals_(iat)[iq] = (s.interpolate().integrate(0) / g - atom_type.zn() * std::exp(-g2 / 4) / g2);
118  }
119  }
120  vloc_radial_integrals_(iat).interpolate();
121  }

150  inline void generate_aug_radial_integrals()
151  {
152  PROFILE("sirius::Radial_integrals::generate_aug_radial_integrals");
153 
154  int nmax = unit_cell_.max_mt_radial_basis_size();
155  int lmax = unit_cell_.lmax();
156 
157  /* interpolate <j_{l_n}(q*x) | Q_{xi,xi'}^{l}(x) > with splines */
158  aug_radial_integrals_ = mdarray<Spline<double>, 3>(nmax * (nmax + 1) / 2, 2 * lmax + 1, unit_cell_.num_atom_types());
159  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
160  auto& atom_type = unit_cell_.atom_type(iat);
161 
162  if (!atom_type.pp_desc().augment) {
163  continue;
164  }
165 
166  /* number of radial beta-functions */
167  int nbrf = atom_type.mt_radial_basis_size();
168  /* maximum l of beta-projectors */
169  int lmax_beta = atom_type.indexr().lmax();
170 
171  for (int l = 0; l <= 2 * lmax_beta; l++) {
172  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
173  aug_radial_integrals_(idx, l, iat) = Spline<double>(grid_gmax_);
174  }
175  }
176 
177  /* interpolate Q-operator radial functions */
178  mdarray<Spline<double>, 2> qrf_spline(nbrf * (nbrf + 1) / 2, 2 * lmax_beta + 1);
179  #pragma omp parallel for
180  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
181  for (int l3 = 0; l3 <= 2 * lmax_beta; l3++) {
182  qrf_spline(idx, l3) = Spline<double>(atom_type.radial_grid());
183  for (int ir = 0; ir < atom_type.num_mt_points(); ir++) {
184  qrf_spline(idx, l3)[ir] = atom_type.pp_desc().q_radial_functions_l(ir, idx, l3);
185  }
186  qrf_spline(idx, l3).interpolate();
187  }
188  }
189 
190  #pragma omp parallel for
191  for (int iq = 0; iq < grid_gmax_.num_points(); iq++) {
192  Spherical_Bessel_functions jl(2 * lmax_beta, atom_type.radial_grid(), grid_gmax_[iq]);
193 
194  for (int l3 = 0; l3 <= 2 * lmax_beta; l3++) {
195  for (int idxrf2 = 0; idxrf2 < nbrf; idxrf2++) {
196  int l2 = atom_type.indexr(idxrf2).l;
197  for (int idxrf1 = 0; idxrf1 <= idxrf2; idxrf1++) {
198  int l1 = atom_type.indexr(idxrf1).l;
199 
200  int idx = idxrf2 * (idxrf2 + 1) / 2 + idxrf1;
201 
202  if (l3 >= std::abs(l1 - l2) && l3 <= (l1 + l2) && (l1 + l2 + l3) % 2 == 0) {
203  aug_radial_integrals_(idx, l3, iat)[iq] = sirius::inner(jl[l3], qrf_spline(idx, l3), 0,
204  atom_type.num_mt_points());
205  }
206  }
207  }
208  }
209  }
210 
211  for (int l = 0; l <= 2 * lmax_beta; l++) {
212  for (int idx = 0; idx < nbrf * (nbrf + 1) / 2; idx++) {
213  aug_radial_integrals_(idx, l, iat).interpolate();
214  }
215  }
216  }

219  inline void generate_beta_radial_integrals()
220  {
221  PROFILE("sirius::Radial_integrals::generate_beta_radial_integrals");
222 
223  /* create space for <j_l(qr)|beta> radial integrals */
224  beta_radial_integrals_ = mdarray<Spline<double>, 2>(unit_cell_.max_mt_radial_basis_size(), unit_cell_.num_atom_types());
225 
226  for (int iat = 0; iat < unit_cell_.num_atom_types(); iat++) {
227  auto& atom_type = unit_cell_.atom_type(iat);
228  int nrb = atom_type.mt_radial_basis_size();
229 
230  for (int idxrf = 0; idxrf < atom_type.mt_radial_basis_size(); idxrf++) {
231  beta_radial_integrals_(idxrf, iat) = Spline<double>(grid_gkmax_);
232  }
233 
234  /* interpolate beta radial functions */
235  std::vector<Spline<double>> beta_rf(nrb);
236  for (int idxrf = 0; idxrf < nrb; idxrf++) {
237  beta_rf[idxrf] = Spline<double>(atom_type.radial_grid());
238  int nr = atom_type.pp_desc().num_beta_radial_points[idxrf];
239  for (int ir = 0; ir < nr; ir++) {
240  beta_rf[idxrf][ir] = atom_type.pp_desc().beta_radial_functions(ir, idxrf);
241  }
242  beta_rf[idxrf].interpolate();
243  }
244 
245  #pragma omp parallel for
246  for (int iq = 0; iq < grid_gkmax_.num_points(); iq++) {
247  Spherical_Bessel_functions jl(unit_cell_.lmax(), atom_type.radial_grid(), grid_gkmax_[iq]);
248  for (int idxrf = 0; idxrf < atom_type.mt_radial_basis_size(); idxrf++) {
249  int l = atom_type.indexr(idxrf).l;
250  int nr = atom_type.pp_desc().num_beta_radial_points[idxrf];
251  /* compute \int j_l(q * r) beta_l(r) r^2 dr */
252  /* remeber that beta(r) are defined as miltiplied by r */
253  beta_radial_integrals_(idxrf, iat)[iq] = sirius::inner(jl[l], beta_rf[idxrf], 1, nr);
254  }
255  }
256  for (int idxrf = 0; idxrf < atom_type.mt_radial_basis_size(); idxrf++) {
257  beta_radial_integrals_(idxrf, iat).interpolate();
258  }
259  }

302  }

306  inline std::pair<int, double> iqdq_gkmax(double q__) const
307  {
308  std::pair<int, double> result;
309  result.first = static_cast<int>((grid_gkmax_.num_points() - 1) * q__ / param_.gk_cutoff());
310  /* delta q = q - q_i */
311  result.second = q__ - grid_gkmax_[result.first];
312  return std::move(result);
313  }
